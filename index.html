function redirect() { // 重新取得當前的寬高 const w = window.innerWidth; const h = window.innerHeight; // 判斷橫式或直式 const isLandscape = w > h; const target = isLandscape ? "index(橫式適用).html" : "index(直式適用).html"; // 檢查目前是否已經在正確的頁面，避免無限迴圈跳轉 if (!window.location.pathname.includes(target)) { window.location.replace(target + "?t=" + new Date().getTime()); } } // 監聽視窗大小改變，這比 orientationchange 更可靠 window.addEventListener("resize", () => { clearTimeout(window.resizeTimer); window.resizeTimer = setTimeout(redirect, 200); }); window.onload = redirect;
